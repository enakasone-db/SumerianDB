SELECT COUNT(o.id_ordenes) as total
FROM public."ordenes" o
JOIN public."detalle_de_ordenes" d ON o.id_ordenes = d.orden_id
JOIN public."compradores" comp ON o.comprador = comp.id_comprador
JOIN public."condicion_op" cond ON d.condicion = cond.id_condicion
JOIN public."territorios" pais ON o.territorio = pais.id
JOIN public."productos" p ON d.producto = p.id_producto
WHERE d.condicion = 1

-- 🔍 1. BLOQUE FILTRO GENERAL (Todo lo relacionado al texto va aquí dentro)
AND (
    COALESCE({{ txt_buscar.text }}, '') = '' 
    OR (
        o.orden ILIKE {{ '%' + txt_buscar.text + '%' }} OR
        comp.comprador ILIKE {{ '%' + txt_buscar.text + '%' }} OR
        pais.territorio ILIKE {{ '%' + txt_buscar.text + '%' }} OR
        p.producto ILIKE {{ '%' + txt_buscar.text + '%' }}
    )
)
